import { getOrderDetail, updateOrderStatus, cancelOrder as cancelOrderApi } from '@/api/order'
import { useOrderStore } from '@/store/order'
import { useRoute } from 'vue-router'

export function useOrderDetail() {
  const route = useRoute()
  const store = useOrderStore()
  const loading = ref(false)

  const fetchData = async () => {
    loading.value = true
    try {
      // BUSINESS LOGIC: 可自定义请求参数

      console.error('[OrderDetail] fetchData error:', e)
    } finally {
      loading.value = false
    }
  }

  onMounted(fetchData)

  return {
    orderInfo: store.orderInfo,

## 步骤三：自动代码修正机制

### 3.1 Pre-commit Hook（自动触发）

`.husky/pre-commit` 自动执行：
1. `eslint --fix` — 修正格式问题
2. `prettier --write` — 统一代码风格
3. `node scripts/validator.js` — 工厂规则校验

### 3.2 工厂自定义 ESLint 规则（`rules/fe-factory-rules.js`）

| 规则 | 说明 | Auto-fix |
|------|------|---------|
| `no-inline-style` | 禁止行内 style，必须用 class | ✅ |
| `require-data-track-id` | 可交互元素必须有 `data-track-id` | ❌需手填 |
| `no-direct-store-mutation` | 禁止直接修改 store，必须通过 action | ✅ |
| `require-loading-state` | 异步函数必须有 loading 状态 | ❌需手填 |
| `no-magic-api-url` | 禁止硬编码 API 路径 | ✅ |
| `require-error-boundary` | 页面必须有 error 处理 | ❌需手填 |

### 3.3 智能代码审查（AI Code Review）

提交 PR 时自动触发，检查点：
- 变量命名是否符合规范（camelCase/PascalCase）
- 是否遗漏 Error Boundary
- 是否重复定义全局变量
- API 调用是否带有 loading/error 状态处理
- 埋点 track-id 是否与 Schema 中定义一致
